              function getBoard() {

                var size = 8;
                var board = "";
                for (var y = 0; y < size; y++) {
                  for (var x = 0; x < size; x++) {
                    if ((x + y) % 2 == 0)
                      board += " ";
                    else
                      board += "#";
                  }
                  board += "\n" + "
";

                  //  board = board + "

                  //  console.log(board);
                }
                // write the Chessboard to the screen
                $("#q3result").html(board);
              }
